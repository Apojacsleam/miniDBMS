     studentNo CHAR(8) PRIMARY KEY NOT NULL,
     studentName CHAR(10),
     grade INT
);
CREATE TABLE course (
     courseNo CHAR(4) PRIMARY KEY NOT NULL,
     courseName CHAR(20),
     credit INT
);
CREATE TABLE score (
     studentNo CHAR(8) PRIMARY KEY NOT NULL,
     courseNo CHAR(4) PRIMARY KEY NOT NULL,
     score INT NOT NULL,
);
